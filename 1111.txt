    def spatial_pooling(self, response_diff):
        """
        執行空間池化操作
        使用 Minkowski 距離計算總體視覺差異
        """
        # 計算 Minkowski 範數
        pooled_response = np.power(
            np.mean(np.power(np.abs(response_diff), self.exponent)), 
            1/self.exponent
        )
        
        return pooled_response
