void SliceSetting::CheckValue(SliceSetting ^ tmpConfig)
{
    this->Name = tmpConfig->Name;

    // 首先调整大小
    if (tmpConfig->ZoneList->Length > 0)
        Array::Resize(this->ZoneList, tmpConfig->ZoneList->Length);
    
    // 使用tmpConfig->ZoneList->Length作为循环条件
    for (int cnt = 0; cnt < tmpConfig->ZoneList->Length; cnt++)
    {
        if (!this->ZoneList[cnt])
            this->ZoneList[cnt] = gcnew ZoneSetting();

        this->ZoneList[cnt]->CheckValue(tmpConfig->ZoneList[cnt]);
        this->ZoneList[cnt]->Index = cnt;
    }
}

void TaskManager::FinishTask(TaskData ^ task)
{
	int index = -1;

	for (int cnt = 0; cnt < this->taskList->Count; cnt++)
	{
		TaskData ^ temp = (TaskData ^) this->taskList[cnt];

		if (temp->Index == task->Index)
		{
			index = cnt;
			break;
		}
	}
	if (index == -1) return;
    	// 添加安全检查
   	if (index == -1 || index >= this->taskList->Count) 
            return;
	this->taskList->RemoveAt(index);
}
